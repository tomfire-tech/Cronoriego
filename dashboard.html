───────────────────────────────────────────────────────────── # LIMPIEZA DE VALORES # ───────────────────────────────────────────────────────────── def clean_text(v): if v is None or v == '': return None if isinstance(v, float) and v.is_integer(): return str(int(v)) return str(v).strip() def clean_num(v): if v is None or v == '': return None if isinstance(v, (int, float)): return v s = str(v).strip() if s == '': return None has_comma, has_dot = ',' in s, '.' in s if has_comma and has_dot: s = s.replace('.', '').replace(',', '.') elif has_comma: s = s.replace(',', '.') try: return float(s) except ValueError: return None def clean_fecha(v): if v is None or v == '': return None if isinstance(v, (int, float)): base = dt.datetime(1899, 12, 30) return (base + dt.timedelta(days=float(v))).date().isoformat() if isinstance(v, dt.datetime): return v.date().isoformat() if isinstance(v, dt.date): return v.isoformat() s = str(v).strip() for fmt in ('%Y-%m-%d', '%d-%m-%Y', '%d/%m/%Y', '%m/%d/%Y'): try: return dt.datetime.strptime(s[:10], fmt).date().isoformat() except ValueError: continue return s def build_record(row, colmap): rec = {} for header, col in colmap.items(): raw = row.get(header) if col in DATE_COLS: rec[col] = clean_fecha(raw) elif col in TEXT_COLS: rec[col] = clean_text(raw) else: rec[col] = clean_num(raw) return rec def chunked(seq, n): for i in range(0, len(seq), n): yield seq[i:i + n] # ───────────────────────────────────────────────────────────── # CARGA DE UN JOB # ───────────────────────────────────────────────────────────── def run_job(name, spec, sh, sb, dry, legacy_upsert=False): print(f"\n=== {name.upper()} ({spec['tab']} -> {spec['table']}) ===") ws = sh.worksheet(spec['tab']) registros = ws.get_all_records(value_render_option='UNFORMATTED_VALUE') print(f" filas crudas: {len(registros)}") keys = spec['keys'] recs = [] for row in registros: rec = build_record(row, spec['map']) if all(rec.get(k) not in (None, '') for k in keys): recs.append(rec) # de-dup por clave lógica (por si el Sheet trae una fila repetida) dedup = {} for rec in recs: dedup[tuple(rec[k] for k in keys)] = rec out = list(dedup.values()) print(f" con datos: {len(recs)} | tras dedup {keys}: {len(out)} " f"| duplicados: {len(recs) - len(out)}") # valores de scope presentes en el Sheet (lo que vamos a "reemplazar") scope = spec['scope'] scope_vals = sorted({rec.get(scope) for rec in out if rec.get(scope) not in (None, '')}) print(f" scope '{scope}': {len(scope_vals)} valores " f"({', '.join(map(str, scope_vals[:8]))}{'…' if len(scope_vals) > 8 else ''})") if out: import json print(" ejemplo:", json.dumps(out[len(out) // 2], ensure_ascii=False)[:300]) if dry: if legacy_upsert: print(" [DRY-RUN] modo legacy-upsert: haría upsert de " f"{len(out)} filas (sin borrar).") else: print(f" [DRY-RUN] BORRARÍA en {spec['table']} las filas con " f"{scope} en esos {len(scope_vals)} valores, y luego " f"INSERTARÍA {len(out)} filas.") return # ── MODO LEGACY (fallback): solo upsert, comportamiento antiguo ── if legacy_upsert: total = 0 for chunk in chunked(out, BATCH): sb.table(spec['table']).upsert(chunk, on_conflict=spec['conflict']).execute() total += len(chunk) print(f" upsert {total}/{len(out)}") print(f" OK (legacy upsert): {total} filas en {spec['table']}") return # ── MODO SINCRONIZACIÓN (delete-and-insert por scope) ── # 1) Borrar en Supabase todo lo de los valores de scope que trae el Sheet. # Se hace por lotes con .in_() para no exceder límites de URL/consulta. if scope_vals: borrados = 0 for grp in chunked(scope_vals, DELETE_IN_CHUNK): sb.table(spec['table']).delete().in_(scope, grp).execute() borrados += len(grp) print(f" delete scope {borrados}/{len(scope_vals)}") else: print(" (sin valores de scope; no se borra nada)") # 2) Insertar limpio lo que trae el Sheet. total = 0 for chunk in chunked(out, BATCH): sb.table(spec['table']).insert(chunk).execute() total += len(chunk) print(f" insert {total}/{len(out)}") print(f" OK (sync): {spec['table']} quedó con {total} filas para este scope") # ───────────────────────────────────────────────────────────── def main(): ap = argparse.ArgumentParser() ap.add_argument('--dry-run', action='store_true', help='Lee y muestra, no escribe') ap.add_argument('--only', choices=list(JOBS.keys()), help='Carga solo un job') ap.add_argument('--legacy-upsert', action='store_true', help='Fallback: modo viejo (solo upsert, sin borrar). No recomendado.') args = ap.parse_args() sa_path = os.environ.get('GOOGLE_SA_JSON') if not sa_path or not os.path.exists(sa_path): sys.exit("ERROR: define GOOGLE_SA_JSON con la ruta al .json del service account.") gc = gspread.service_account(filename=sa_path) sh = gc.open_by_key(SHEET_ID) sb = None if not args.dry_run: url = os.environ.get('SUPABASE_URL') key = os.environ.get('SUPABASE_SERVICE_KEY') if not url or not key: sys.exit("ERROR: define SUPABASE_URL y SUPABASE_SERVICE_KEY (service_role).") sb = create_client(url, key) jobs = {args.only: JOBS[args.only]} if args.only else JOBS for name, spec in jobs.items(): run_job(name, spec, sh, sb, args.dry_run, legacy_upsert=args.legacy_upsert) print("\nListo." + (" (dry-run, sin escribir)" if args.dry_run else "")) if __name__ == '__main__': main()
